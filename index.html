sales_analysis.sql 
WITH monthly_sales AS (
    SELECT 
        DATE_TRUNC('month', order_date) AS sale_month,
        SUM(total_amount) AS revenue,
        COUNT(order_id) AS total_orders
    FROM ecom_orders
    WHERE status = 'Completed'
    GROUP BY 1
)
SELECT 
    sale_month,
    revenue,
    revenue - LAG(revenue) OVER (ORDER BY sale_month) AS revenue_diff,
    ROUND(100.0 * (revenue - LAG(revenue) OVER (ORDER BY sale_month)) 
        / LAG(revenue) OVER (ORDER BY sale_month), 2) AS growth_rate
FROM monthly_sales;
